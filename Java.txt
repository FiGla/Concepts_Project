int  fabo (int n){ 
	if( n  < = 1){
		return 1;
	}
	return n*fabo(n-1);
	}

void  Reverse (int x){ 
	int y=5;
	for(int  i=0 ; i<x.size(); ++i ){
		if( x  <  y ){
			y=x;
		}
	}
	return y;
}
